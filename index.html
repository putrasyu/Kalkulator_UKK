                const screen = document.getElementById('screen');
                const operationDisplay = document.createElement('div');
                operationDisplay.style.textAlign = 'right';
                operationDisplay.style.fontWeight = 'bold';
                operationDisplay.style.marginBottom = '5px';
                screen.parentNode.insertBefore(operationDisplay, screen);

                let numbers = [], operators = [], isResultDisplayed = false;

                function addInput(val) {
                    if (isResultDisplayed) return;
                    if ('+-*/'.includes(val)) {
                        numbers.push(parseFloat(screen.value));
                        operators.push(val);
                        operationDisplay.textContent +=  \${screen.value} \${val};
                        screen.value = '';
                    } else {
                        screen.value += val;
                        screen.scrollLeft = screen.scrollWidth;
                    }
                    screen.style.opacity = 0;
                    setTimeout(() => screen.style.opacity = 1, 50);
                }

                function clearScreen() {
                    screen.value = '';
                    numbers = [];
                    operators = [];
                    operationDisplay.textContent = '';
                    isResultDisplayed = false;
                }

                function deleteLast() {
                    if (isResultDisplayed) return;
                    if (!screen.value && numbers.length) {
                        operationDisplay.textContent = operationDisplay.textContent.trim().split(' ').slice(0, -2).join(' ');
                        numbers.pop(); operators.pop();
                    } else {
                        screen.value = screen.value.slice(0, -1);
                    }
                }

                function calculate() {
                    if (!screen.value || !operators.length) return;

                    numbers.push(parseFloat(screen.value));
                    let r = numbers[0];

                    for (let i = 0; i < operators.length; i++) {
                        let n = numbers[i + 1];
                        let o = operators[i];

                        if (o === '+') {
                            r = r + n;
                        } else if (o === '-') {
                            r = r - n;
                        } else if (o === '*') {
                            r = r * n;
                        } else if (o === '/') {
                            r = n !== 0 ? r / n : r; // hindari pembagian dengan nol
                        }
                    }

                    operationDisplay.textContent +=  \${screen.value} =;
                    screen.value = r;
                    numbers = [r];
                    operators = [];
                    isResultDisplayed = true;
                }

                document.addEventListener('keydown', e => {
                    if ('0123456789.'.includes(e.key)) addInput(e.key);
                    if ('+-*/'.includes(e.key)) addInput(e.key);
                    if (e.key === 'Enter') calculate();
                    if (e.key === 'Delete') clearScreen();
                    if (e.key === 'Backspace') deleteLast();
                });

                function theme() {
                    document.body.classList.toggle('light-mode');
                    screen.classList.toggle('light-mode');
                    document.querySelector('.calculator').classList.toggle('light-mode');
                    document.querySelector('.instructions').classList.toggle('light-mode');
                }
